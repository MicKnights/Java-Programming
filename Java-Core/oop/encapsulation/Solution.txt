class Default{
    int number = 0;
    String word = "zero";

    public void print(int number, String word){
        System.out.println("number equal: "+number+" and word is: "+word);
    }

}

class Public{
    public int number = 1;
    public String word = "one";

    public void print(int number, String word){
        System.out.println("number equal: "+number+" and word is: "+word);
    }

}
 

class Protected{
    protected int number = 2;
    protected String word = "two";

    public void print(int number, String word){
        System.out.println("number equal: "+number+" and word is: "+word);
    }

}

class Private{
    private int number = 3;
    private String word = "three";

    public void print(int number, String word){
        System.out.println("number equal: "+number+" and word is: "+word);
    }

}
